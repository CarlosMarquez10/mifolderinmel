CREATE TABLE empleados (
    proyecto_sede VARCHAR(50),
    num_cedula BIGINT PRIMARY KEY,
    nombres VARCHAR(130),
    apellidos VARCHAR(130),
    eps VARCHAR(50),
    fondo_pension VARCHAR(120),
    licencia VARCHAR(5),
    categoria_licencia VARCHAR(30),
    num_licencia_conduccion BIGINT,
    direccion_empleado VARCHAR(150),
    barrio_sector VARCHAR(50),
    ciudad_municipio VARCHAR(50),
    telefono_personal VARCHAR(20),
    celular_personal VARCHAR(20),
    fecha_nacimiento VARCHAR(50),
    lugar_nacimiento VARCHAR(100),
    rh VARCHAR(5),
    num_hijos INT,
    estado_civil VARCHAR(30),
    nivel_escolaridad VARCHAR(100),
    nombre_titulo_aprobado VARCHAR(150),
    num_matricula_profesional VARCHAR(50),
    talla_camisa VARCHAR(5),
    talla_pantalon INT,
    talla_zapatos INT,
    nombre_conyuge VARCHAR(130),
    num_cedula_conyuge BIGINT,
    fecha_nacimiento_conyuge VARCHAR(50),
    conte VARCHAR(5),
    email VARCHAR(130),
    es_papa_si_no VARCHAR(2)
);
