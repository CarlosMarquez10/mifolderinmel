CREATE TABLE documentos_empleados (
    id INT AUTO_INCREMENT PRIMARY KEY,
    cedula BIGINT NOT NULL,
    fecha_registro DATE,
    certificado_eps VARCHAR(2),
    certificado_afp VARCHAR(2),
    certificados_estudio VARCHAR(2),
    certificado_diploma_bachiller VARCHAR(2),
    certificado_titulo_profesional VARCHAR(2),
    tarjeta_conte VARCHAR(2),
    certificacion_laboral VARCHAR(2),
    hoja_vida VARCHAR(2),
    cedula_empleado VARCHAR(2),
    matricula_profesional VARCHAR(2),
    carne_vacunacion VARCHAR(2),
    antecedentes_judiciales VARCHAR(2),
    contraloria VARCHAR(2),
    policia VARCHAR(2),
    procuraduria VARCHAR(2),
    certificacion_manejo_defensivo VARCHAR(2),
    formacion_primeros_auxilios VARCHAR(2),
    carne_vacunacion_covid VARCHAR(2),
    rut_actividad_economica_0010 VARCHAR(2),
    foto_perfil VARCHAR(2),
    EstadoInfo VARCHAR(20),
    EstadoAnexos VARCHAR(20)
);
